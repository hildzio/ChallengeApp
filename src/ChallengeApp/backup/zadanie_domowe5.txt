/*Dzień 5: Tablice, Listy i Klasy ; Zadanie domowe */
using System;
using System.Collections.Generic;

namespace ChallengeApp
{
    class Program
    {
        static void Main(string[] args)
        {
            var productName = new List<string>(2);
            productName.Add("Masło");
            productName.Add("Chleb");

            var price = new List<int>(3);
            price.Add(6);
            price.Add(7);
            price.Add(4);

            foreach (var n in productName)
            {
                Console.WriteLine(n);
                foreach (var m in price)
                {
                    Console.WriteLine(m);
                }
            }
        }
    }
}
